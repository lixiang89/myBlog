所以其实就是找数字字符串中的最大数字。


```javascript
/**
 * @param {string} n
 * @return {number}
 */
var minPartitions = function(n) {
    var max=0
    for(var i=0;i<n.length;i++){
        var t=n[i]
        if(t==9) return 9
        if(max<t){
            max=t
        }
    }
    return max
};
```
