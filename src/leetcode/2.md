[两数相加](https://leetcode-cn.com/problems/add-two-numbers/)

消耗：

![消耗](/images/leetcode/2.png)

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function(l1, l2) {
    var temp=new ListNode()
    var res=temp
    var plus=0// 记录是否要加1
    while(l1||l2){
        var l1Val=0
        var l2Val=0
        var l1Next
        var l2Next
        if(l1){
            l1Val=l1.val
            l1Next=l1.next
        }
        if(l2){
            l2Val=l2.val
            l2Next=l2.next
        }

        var val=l1Val+l2Val+plus
        if(val>9){//val 大于9的话，就赋值为个位数，plus赋值为1
            val=val%10
            plus=1
        }else{//不大于9 plus为0，因为上面代码已经加过plus了，所以没问题
            plus=0
        }
        temp.val=val
        //当l1或l2有next的时候，这里也要创建next，当走到这里plus还是1的话，说明这是最后加完后需要进位，所以也需要创建next，其他情况无需next
        temp.next=(l1Next||l2Next||plus)?new ListNode(plus):null
        // temp，l1，l2赋值为其子节点，继续循环
        temp=temp.next
        l1=l1Next
        l2=l2Next
    }
    return res
};
```
