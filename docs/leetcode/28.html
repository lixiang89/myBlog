    /**
 * @param {string} haystack
 * @param {string} needle
 * @return {number}
 */
var strStr = function(haystack, needle) {
    if(haystack===needle) return 0
    if(haystack==="") return -1
    if(needle==="") return 0
    if(needle.length>haystack.length) return -1
    var j=0,i=0,res=-1

    while(i>=0){
        var c=haystack[i]
        var len=needle.length
        if(j===len) break
        if(i==haystack.length){
            if(j<len){
                res=-1
            }
            break
        }

        if(c===needle[j]){
            j++
            if(res===-1) res=i
            i++
        }else{
            if(res>-1){
                i=res+1
                j=0
                res=-1
            }else{
                i++
            }
            
        }
    }
  
    return res
};
